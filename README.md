# grading
grade generating exercise

grade = int(input ("What is your grade?\n:"))

if grade >= 0 and grade < 59:
    print("F")
if grade >= 60 and grade < 69:
    print("D")
if grade >= 70 and grade < 79:
    print("C")
if grade >= 80 and grade < 89:
    print ("B")
if grade >= 90 and grade <=100:
    print ("A")
